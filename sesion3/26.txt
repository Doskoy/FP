count == 0 && limit < 20    					->  true	

limit > 20 || count < 5     					->  true

!(count == 12)   								->	true
	
count == 1 && x < y  							-> 	false
	
!((count < 10 || x < y) && count >= 0) 			-> 	false

(count > 5 && y == 7) || (count <= 0 && limit = 5*x) -> true

!( limit != 10 && x > y) 						-> true
